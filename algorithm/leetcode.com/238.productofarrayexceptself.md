# 238.ProductOfArrayExceptSelf

```python
from functools import reduce

class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        product = lambda x, y: x*y
        result = []
        if 0 in nums:
            # N*(N-1)
            for index, number in enumerate(nums):
                result.append(reduce(product, nums[:index]+nums[index+1:]))
        else:
            # N + 1 + N
            result = [reduce(product, nums)]
            result = result * len(nums)
            for index, number in enumerate(nums):
                result[index] = result[index] // number
        return result
```
